### **Weekly Distribution Logic**
```python
def distribute_across_week(tasks):
    weekly_plan = {
        "monday": [],
        "tuesday": [], 
        "wednesday": [],
        "thursday": [],
        "friday": []
    }
    
    # Distribute so each day has manageable workload
    for day in weekly_plan:
        daily_tasks = []
        time_budget = 6 * 60  # 6 hours in minutes
        
        # Add urgent tasks first
        urgent_tasks = get_urgent_for_day(day)
        daily_tasks.extend(urgent_tasks)
        time_budget -= sum(task.time_needed for task in urgent_tasks)
        
        # Fill remaining time with medium priority
        medium_tasks = get_medium_priority()
        for task in medium_tasks:
            if time_budget >= task.time_needed:
                daily_tasks.append(task)
                time_budget -= task.time_needed
        
        weekly_plan[day] = daily_tasks
    
    return weekly_plan
```
